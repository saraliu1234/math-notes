\begin{definition}
    Let $X\subseteq \mathcal{M}$ with $|X|=k$.
    $S\subseteq N$ is a \textbf{blocking coalition} against $X$ if $|S|\geq\lceil \frac{n}{k}\rceil$ and $\exists y\in\mathcal{M}$ such that $\forall i\in S$, $d(i,y)<D_i(X)$.
    $X\subseteq \mathcal{N}$ is \textbf{proportional} if there is no blocking coalition against $X$.
\end{definition}
